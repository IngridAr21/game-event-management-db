// PLAYERS PARTICIPATE IN QUESTS
// Create 15 PARTICIPATED_IN relationships with the new quest names
MATCH (p:Player), (q:Quest)
WHERE q.name IN [
  'Retrieve the Lost Artifact',
  'Defend the Village',
  'Uncover the Secret of the Forbidden Temple',
  'Rescue the Trapped Miners',
  'Banish the Ancient Curse'
]
WITH p, q
LIMIT 15
CREATE (p)-[:PARTICIPATED_IN {
  itemsCollected: CASE
    WHEN q.name = 'Retrieve the Lost Artifact' THEN ['Sword', 'Potion']
    WHEN q.name = 'Defend the Village' THEN ['Bow', 'Arrows']
    WHEN q.name = 'Uncover the Secret of the Forbidden Temple' THEN ['Shield', 'Magic Staff']
    WHEN q.name = 'Rescue the Trapped Miners' THEN ['Potion of Healing', 'Bow']
    WHEN q.name = 'Banish the Ancient Curse' THEN ['Magic Staff', 'Sword']
    ELSE []
  END,
  pointsScored: CASE
    WHEN q.name = 'Retrieve the Lost Artifact' THEN 100
    WHEN q.name = 'Defend the Village' THEN 150
    WHEN q.name = 'Uncover the Secret of the Forbidden Temple' THEN 200
    WHEN q.name = 'Rescue the Trapped Miners' THEN 120
    WHEN q.name = 'Banish the Ancient Curse' THEN 180
    ELSE 0
  END,
  timeTaken: CASE
    WHEN q.name = 'Retrieve the Lost Artifact' THEN '2 hours'
    WHEN q.name = 'Defend the Village' THEN '3 hours'
    WHEN q.name = 'Uncover the Secret of the Forbidden Temple' THEN '4 hours'
    WHEN q.name = 'Rescue the Trapped Miners' THEN '2.5 hours'
    WHEN q.name = 'Banish the Ancient Curse' THEN '3.5 hours'
    ELSE '0 hours'
  END
}]->(q);
